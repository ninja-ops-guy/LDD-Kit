# Example: LDD-Kit with Go / Gin

> Complete minimal example of using LDD-Kit with a Go Gin web service.

## Prerequisites

- Go 1.21+
- `go` CLI

## Project Setup

```bash
# 1. Create project directory
mkdir go-gin-example && cd go-gin-example

# 2. Initialize Go module
go mod init go-gin-example

# 3. Install dependencies
go get github.com/gin-gonic/gin
go get go.opentelemetry.io/otel
go get go.opentelemetry.io/otel/sdk
go get go.opentelemetry.io/otel/exporters/otlp/otlptrace/otlptracegrpc
go get go.opentelemetry.io/contrib/instrumentation/github.com/gin-gonic/gin/otelgin
go get github.com/prometheus/client_golang/prometheus
go get github.com/prometheus/client_golang/prometheus/promhttp
```

## Files

### `main.go` — Complete Minimal Gin App with Telemetry

```go
// Minimal Gin application with LDD-Kit telemetry.
//
// This demonstrates:
//   - Telemetry setup and initialization
//   - Telemetry middleware for automatic HTTP request tracing
//   - Structured logging with event schemas
//   - Prometheus metrics endpoint
//   - Trace context propagation
//
// Environment variables:
//     MY_SERVICE_TELEMETRY_ENABLED=true|false
//     MY_SERVICE_LOG_FORMAT=json|console
//     OTEL_SERVICE_NAME=my-service
//     OTEL_TRACES_SAMPLER=always_on|parentbased_traceidratio

package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"os"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/prometheus/client_golang/prometheus"
	"github.com/prometheus/client_golang/prometheus/promhttp"

	"go.opentelemetry.io/otel"
	"go.opentelemetry.io/otel/attribute"
	"go.opentelemetry.io/otel/exporters/otlp/otlptrace/otlptracegrpc"
	"go.opentelemetry.io/otel/sdk/resource"
	sdktrace "go.opentelemetry.io/otel/sdk/trace"
	semconv "go.opentelemetry.io/otel/semconv/v1.21.0"
	"go.opentelemetry.io/otel/trace"
	"go.opentelemetry.io/contrib/instrumentation/github.com/gin-gonic/gin/otelgin"
)

const serviceName = "go-gin-example"

// =============================================================================
// Configuration
// =============================================================================
var (
	telemetryEnabled = getEnv("MY_SERVICE_TELEMETRY_ENABLED", "false") == "true"
	logFormat        = getEnv("MY_SERVICE_LOG_FORMAT", "console")
	otelServiceName  = getEnv("OTEL_SERVICE_NAME", serviceName)
)

func getEnv(key, fallback string) string {
	if v := os.Getenv(key); v != "" {
		return v
	}
	return fallback
}

// =============================================================================
// Structured Logger (generated by LDD-Kit adapt.py)
// =============================================================================
type Logger struct {
	service string
	format  string // "json" or "console"
}

func NewLogger(service, format string) *Logger {
	return &Logger{service: service, format: format}
}

func (l *Logger) Info(ctx context.Context, event string, fields ...interface{}) {
	if l.format == "json" {
		entry := map[string]interface{}{
			"event_type": event,
			"service":    l.service,
			"timestamp":  time.Now().UTC().Format(time.RFC3339),
			"level":      "info",
		}
		// Add fields
		for i := 0; i < len(fields)-1; i += 2 {
			key, ok := fields[i].(string)
			if ok {
				entry[key] = fields[i+1]
			}
		}
		// Add trace info
		span := trace.SpanFromContext(ctx)
		if span.SpanContext().IsValid() {
			entry["trace_id"] = span.SpanContext().TraceID().String()
			entry["span_id"] = span.SpanContext().SpanID().String()
		}
		b, _ := json.Marshal(entry)
		fmt.Println(string(b))
	} else {
		// Console format
		span := trace.SpanFromContext(ctx)
		traceInfo := ""
		if span.SpanContext().IsValid() {
			traceInfo = fmt.Sprintf(" [trace=%s]", span.SpanContext().TraceID().String()[:8])
		}
		fieldStr := ""
		for i := 0; i < len(fields)-1; i += 2 {
			fieldStr += fmt.Sprintf(" %v=%v", fields[i], fields[i+1])
		}
		log.Printf("[INFO]%s %s%s", traceInfo, event, fieldStr)
	}
}

func (l *Logger) Error(ctx context.Context, event string, fields ...interface{}) {
	fields = append(fields, "level", "error")
	l.Info(ctx, event, fields...)
}

var logger = NewLogger(serviceName, logFormat)

// =============================================================================
// OpenTelemetry Tracing Setup (generated by LDD-Kit adapt.py)
// =============================================================================
func setupTracing() func() {
	ctx := context.Background()

	res, err := resource.New(ctx,
		resource.WithAttributes(
			semconv.ServiceName(otelServiceName),
			semconv.ServiceVersion("1.0.0"),
		),
	)
	if err != nil {
		log.Fatalf("failed to create resource: %v", err)
	}

	var exporter sdktrace.SpanExporter
	otlpEndpoint := os.Getenv("OTEL_EXPORTER_OTLP_ENDPOINT")
	if otlpEndpoint != "" {
		exp, err := otlptracegrpc.New(ctx, otlptracegrpc.WithEndpointURL(otlpEndpoint))
		if err != nil {
			log.Printf("warning: OTLP exporter failed, falling back to stdout: %v", err)
			exporter = &stdoutExporter{}
		} else {
			exporter = exp
		}
	} else {
		exporter = &stdoutExporter{}
	}

	tp := sdktrace.NewTracerProvider(
		sdktrace.WithSampler(sdktrace.AlwaysSample()),
		sdktrace.WithBatcher(exporter),
		sdktrace.WithResource(res),
	)
	otel.SetTracerProvider(tp)

	return func() {
		if err := tp.Shutdown(ctx); err != nil {
			log.Printf("error shutting down tracer: %v", err)
		}
	}
}

// stdoutExporter is a fallback span exporter for development.
type stdoutExporter struct{}

func (s *stdoutExporter) ExportSpans(ctx context.Context, spans []sdktrace.ReadOnlySpan) error {
	for _, span := range spans {
		fmt.Printf("[TRACE] %s trace_id=%s span_id=%s duration=%s\n",
			span.Name(),
			span.SpanContext().TraceID().String(),
			span.SpanContext().SpanID().String(),
			span.EndTime().Sub(span.StartTime()),
		)
	}
	return nil
}

func (s *stdoutExporter) Shutdown(ctx context.Context) error { return nil }

var tracer = otel.Tracer(serviceName)

// =============================================================================
// Prometheus Metrics (generated by LDD-Kit adapt.py)
// =============================================================================
var (
	httpRequestsTotal = prometheus.NewCounterVec(prometheus.CounterOpts{
		Name: "gin_example_http_requests_total",
		Help: "Total HTTP requests",
	}, []string{"method", "path", "status_code"})

	httpRequestDuration = prometheus.NewHistogramVec(prometheus.HistogramOpts{
		Name:    "gin_example_http_request_duration_seconds",
		Help:    "HTTP request duration in seconds",
		Buckets: prometheus.DefBuckets,
	}, []string{"method", "path"})

	businessEventsTotal = prometheus.NewCounterVec(prometheus.CounterOpts{
		Name: "gin_example_business_events_total",
		Help: "Total business events",
	}, []string{"domain", "event_type"})
)

func init() {
	prometheus.MustRegister(httpRequestsTotal, httpRequestDuration, businessEventsTotal)
}

// =============================================================================
// Event Schemas (generated by LDD-Kit adapt.py)
// =============================================================================
type UserCreated struct {
	EventType  string `json:"event_type"`
	EventDomain string `json:"event_domain"`
	Timestamp  string `json:"timestamp"`
	UserID     string `json:"user_id"`
	Email      string `json:"email"`
	Tier       string `json:"tier"`
}

func NewUserCreated(userID, email, tier string) UserCreated {
	return UserCreated{
		EventType:   "user.created",
		EventDomain: "user",
		Timestamp:   time.Now().UTC().Format(time.RFC3339),
		UserID:      userID,
		Email:       email,
		Tier:        tier,
	}
}

func (e UserCreated) ToFields() []interface{} {
	return []interface{}{
		"user_id", e.UserID,
		"email", e.Email,
		"tier", e.Tier,
		"event_domain", e.EventDomain,
	}
}

type UserDeleted struct {
	EventType   string `json:"event_type"`
	EventDomain string `json:"event_domain"`
	Timestamp   string `json:"timestamp"`
	UserID      string `json:"user_id"`
	Reason      string `json:"reason"`
}

type PaymentProcessed struct {
	EventType   string `json:"event_type"`
	EventDomain string `json:"event_domain"`
	Timestamp   string `json:"timestamp"`
	PaymentID   string `json:"payment_id"`
	AmountCents int    `json:"amount_cents"`
	Currency    string `json:"currency"`
	Status      string `json:"status"`
}

// =============================================================================
// Telemetry Middleware (generated by LDD-Kit adapt.py)
// =============================================================================
func TelemetryMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		path := c.Request.URL.Path
		method := c.Request.Method

		c.Next()

		duration := time.Since(start)
		statusCode := fmt.Sprintf("%d", c.Writer.Status())

		httpRequestsTotal.WithLabelValues(method, path, statusCode).Inc()
		httpRequestDuration.WithLabelValues(method, path).Observe(duration.Seconds())

		logger.Info(c.Request.Context(), "http_request",
			"method", method,
			"path", path,
			"status_code", c.Writer.Status(),
			"duration_ms", duration.Milliseconds(),
		)
	}
}

// =============================================================================
// Main
// =============================================================================
func main() {
	var shutdown func()
	if telemetryEnabled {
		shutdown = setupTracing()
		defer shutdown()
	}

	// Use release mode in production
	if os.Getenv("GIN_MODE") == "" {
		gin.SetMode(gin.DebugMode)
	}

	router := gin.New()
	router.Use(gin.Recovery())

	// Add OpenTelemetry middleware
	if telemetryEnabled {
		router.Use(otelgin.Middleware(otelServiceName))
	}
	router.Use(TelemetryMiddleware())

	// Health checks
	router.GET("/health", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{"status": "ok", "service": serviceName})
	})

	router.GET("/ready", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{"status": "ready"})
	})

	// Prometheus metrics
	router.GET("/metrics", gin.WrapH(promhttp.Handler()))

	// Business routes
	router.POST("/api/v1/users", handleCreateUser)
	router.DELETE("/api/v1/users/:user_id", handleDeleteUser)
	router.POST("/api/v1/payments", handleProcessPayment)
	router.GET("/api/v1/users/:user_id", handleGetUser)

	port := getEnv("PORT", "8080")
	logger.Info(context.Background(), "service_starting", "port", port)
	router.Run(":" + port)
}

// =============================================================================
// Handlers
// =============================================================================
func handleCreateUser(c *gin.Context) {
	ctx := c.Request.Context()

	var req struct {
		UserID string `json:"user_id"`
		Email  string `json:"email"`
		Tier   string `json:"tier"`
	}
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	if req.Tier == "" {
		req.Tier = "free"
	}

	_, span := tracer.Start(ctx, "create_user")
	span.SetAttributes(
		attribute.String("user.id", req.UserID),
		attribute.String("user.email", req.Email),
	)
	defer span.End()

	logger.Info(ctx, "user.creating", "user_id", req.UserID, "email", req.Email, "tier", req.Tier)

	// Simulate business logic
	time.Sleep(10 * time.Millisecond)

	event := NewUserCreated(req.UserID, req.Email, req.Tier)
	logger.Info(ctx, "user.created", event.ToFields()...)
	businessEventsTotal.WithLabelValues("user", "user.created").Inc()

	c.JSON(http.StatusCreated, gin.H{
		"user_id": req.UserID,
		"status":  "created",
		"tier":    req.Tier,
	})
}

func handleDeleteUser(c *gin.Context) {
	ctx := c.Request.Context()
	userID := c.Param("user_id")

	logger.Info(ctx, "user.deleting", "user_id", userID)

	event := UserDeleted{
		EventType:   "user.deleted",
		EventDomain: "user",
		Timestamp:   time.Now().UTC().Format(time.RFC3339),
		UserID:      userID,
		Reason:      "user_request",
	}
	logger.Info(ctx, "user.deleted",
		"user_id", event.UserID,
		"reason", event.Reason,
	)
	businessEventsTotal.WithLabelValues("user", "user.deleted").Inc()

	c.JSON(http.StatusOK, gin.H{"user_id": userID, "status": "deleted"})
}

func handleProcessPayment(c *gin.Context) {
	ctx := c.Request.Context()

	var req struct {
		PaymentID   string `json:"payment_id"`
		AmountCents int    `json:"amount_cents"`
		Currency    string `json:"currency"`
	}
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	if req.Currency == "" {
		req.Currency = "USD"
	}

	_, span := tracer.Start(ctx, "process_payment")
	span.SetAttributes(
		attribute.String("payment.id", req.PaymentID),
		attribute.Int("payment.amount_cents", req.AmountCents),
	)
	defer span.End()

	logger.Info(ctx, "payment.processing", "payment_id", req.PaymentID, "amount_cents", req.AmountCents)

	// Simulate processing
	time.Sleep(20 * time.Millisecond)

	event := PaymentProcessed{
		EventType:   "payment.processed",
		EventDomain: "payment",
		Timestamp:   time.Now().UTC().Format(time.RFC3339),
		PaymentID:   req.PaymentID,
		AmountCents: req.AmountCents,
		Currency:    req.Currency,
		Status:      "success",
	}
	logger.Info(ctx, "payment.processed",
		"payment_id", event.PaymentID,
		"amount_cents", event.AmountCents,
		"currency", event.Currency,
	)
	businessEventsTotal.WithLabelValues("payment", "payment.processed").Inc()

	c.JSON(http.StatusOK, gin.H{
		"payment_id":   req.PaymentID,
		"status":       "processed",
		"amount_cents": req.AmountCents,
	})
}

func handleGetUser(c *gin.Context) {
	ctx := c.Request.Context()
	userID := c.Param("user_id")

	logger.Info(ctx, "user.fetching", "user_id", userID)

	c.JSON(http.StatusOK, gin.H{
		"user_id": userID,
		"email":   fmt.Sprintf("user_%s@example.com", userID),
		"tier":    "free",
	})
}
```

## Running the Example

```bash
# Development mode (console logs, 100% trace sampling)
export MY_SERVICE_TELEMETRY_ENABLED=true
export MY_SERVICE_LOG_FORMAT=console
export OTEL_SERVICE_NAME=gin-example-local
export OTEL_TRACES_SAMPLER=always_on

go run main.go

# In another terminal, test the endpoints:
curl http://localhost:8080/health
curl -X POST http://localhost:8080/api/v1/users \
  -H "Content-Type: application/json" \
  -d '{"user_id":"usr-001","email":"alice@example.com","tier":"pro"}'
curl http://localhost:8080/metrics
curl -X POST http://localhost:8080/api/v1/payments \
  -H "Content-Type: application/json" \
  -d '{"payment_id":"pay-001","amount_cents":999,"currency":"USD"}'
```

## What to Observe

1. **Structured logs** in the terminal with `event_type`, `trace_id`, and `duration_ms`
2. **Trace spans** for each HTTP request and business operation (printed to stdout in dev mode)
3. **Prometheus metrics** at `/metrics` showing request counts and durations
4. **Event schemas** validated for every business event (`user.created`, `payment.processed`)
