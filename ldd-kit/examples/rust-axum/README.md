# Example: LDD-Kit with Rust / Axum

> Complete minimal example of using LDD-Kit with a Rust Axum web service.

## Prerequisites

- Rust 1.75+ (stable)
- `cargo`

## Project Setup

```bash
# 1. Create project directory
cargo new --bin rust-axum-example && cd rust-axum-example

# 2. Add dependencies to Cargo.toml
```

## Files

### `src/main.rs` — Complete Minimal Axum App with Telemetry

```rust
//! Minimal Axum application with LDD-Kit telemetry.
//!
//! This demonstrates:
//!   - Telemetry setup and initialization
//!   - Telemetry middleware for automatic HTTP request tracing
//!   - Structured logging with event schemas
//!   - Prometheus metrics endpoint
//!   - Trace context propagation
//!
//! Environment variables:
//!     MY_SERVICE_TELEMETRY_ENABLED=true|false
//!     MY_SERVICE_LOG_FORMAT=json|console
//!     OTEL_SERVICE_NAME=my-service
//!     OTEL_TRACES_SAMPLER=always_on|parentbased_traceidratio

use axum::{
    extract::{Path, Request},
    http::StatusCode,
    middleware::{self, Next},
    response::{IntoResponse, Response},
    routing::{delete, get, post},
    Json, Router,
};
use chrono::Utc;
use opentelemetry::{
    global,
    trace::{Span, Tracer},
    Context, KeyValue,
};
use opentelemetry_otlp::WithExportConfig;
use opentelemetry_sdk::{
    propagation::TraceContextPropagator,
    resource::{EnvResourceDetector, Resource, SdkProvidedResourceDetector},
    trace::{self, RandomIdGenerator, Sampler},
    runtime::Tokio,
};
use opentelemetry_semantic_conventions::resource::SERVICE_NAME;
use prometheus::{Counter, Encoder, Histogram, Registry, TextEncoder};
use serde::{Deserialize, Serialize};
use serde_json::json;
use std::{
    env,
    net::SocketAddr,
    sync::Arc,
    time::Instant,
};
use tower_http::trace::TraceLayer;
use tracing::{info, Level};
use tracing_subscriber::{layer::SubscriberExt, util::SubscriberInitExt, EnvFilter};

// =============================================================================
// Configuration
// =============================================================================
const SERVICE_NAME_VALUE: &str = "rust-axum-example";

fn telemetry_enabled() -> bool {
    env::var("MY_SERVICE_TELEMETRY_ENABLED")
        .unwrap_or_else(|_| "false".to_string())
        .to_lowercase()
        == "true"
}

fn log_format() -> String {
    env::var("MY_SERVICE_LOG_FORMAT").unwrap_or_else(|_| "console".to_string())
}

fn otel_service_name() -> String {
    env::var("OTEL_SERVICE_NAME").unwrap_or_else(|_| SERVICE_NAME_VALUE.to_string())
}

// =============================================================================
// Event Schemas (generated by LDD-Kit adapt.py)
// =============================================================================
#[derive(Debug, Serialize)]
struct UserCreated {
    event_type: &'static str,
    event_domain: &'static str,
    timestamp: String,
    user_id: String,
    email: String,
    tier: String,
}

impl UserCreated {
    fn new(user_id: String, email: String, tier: String) -> Self {
        Self {
            event_type: "user.created",
            event_domain: "user",
            timestamp: Utc::now().to_rfc3339(),
            user_id,
            email,
            tier,
        }
    }
}

#[derive(Debug, Serialize)]
struct UserDeleted {
    event_type: &'static str,
    event_domain: &'static str,
    timestamp: String,
    user_id: String,
    reason: Option<String>,
}

impl UserDeleted {
    fn new(user_id: String, reason: Option<String>) -> Self {
        Self {
            event_type: "user.deleted",
            event_domain: "user",
            timestamp: Utc::now().to_rfc3339(),
            user_id,
            reason,
        }
    }
}

#[derive(Debug, Serialize)]
struct PaymentProcessed {
    event_type: &'static str,
    event_domain: &'static str,
    timestamp: String,
    payment_id: String,
    amount_cents: i64,
    currency: String,
    status: String,
}

impl PaymentProcessed {
    fn new(payment_id: String, amount_cents: i64, currency: String, status: String) -> Self {
        Self {
            event_type: "payment.processed",
            event_domain: "payment",
            timestamp: Utc::now().to_rfc3339(),
            payment_id,
            amount_cents,
            currency,
            status,
        }
    }
}

// =============================================================================
// Prometheus Metrics (generated by LDD-Kit adapt.py)
// =============================================================================
#[derive(Clone)]
struct Metrics {
    http_requests_total: CounterVec,
    http_request_duration: HistogramVec,
    business_events_total: CounterVec,
    registry: Registry,
}

use prometheus::{CounterVec, HistogramVec, Opts, HistogramOpts};

impl Metrics {
    fn new() -> Result<Self, prometheus::Error> {
        let registry = Registry::new();

        let http_requests_total = CounterVec::new(
            Opts::new(
                "axum_example_http_requests_total",
                "Total HTTP requests"
            ),
            &["method", "path", "status_code"]
        )?;

        let http_request_duration = HistogramVec::new(
            HistogramOpts::new(
                "axum_example_http_request_duration_seconds",
                "HTTP request duration in seconds"
            ),
            &["method", "path"]
        )?;

        let business_events_total = CounterVec::new(
            Opts::new(
                "axum_example_business_events_total",
                "Total business events"
            ),
            &["domain", "event_type"]
        )?;

        registry.register(Box::new(http_requests_total.clone()))?;
        registry.register(Box::new(http_request_duration.clone()))?;
        registry.register(Box::new(business_events_total.clone()))?;

        Ok(Self {
            http_requests_total,
            http_request_duration,
            business_events_total,
            registry,
        })
    }

    fn record_request(&self, method: &str, path: &str, status_code: u16, duration_secs: f64) {
        self.http_requests_total
            .with_label_values(&[method, path, &status_code.to_string()])
            .inc();
        self.http_request_duration
            .with_label_values(&[method, path])
            .observe(duration_secs);
    }

    fn record_business_event(&self, domain: &str, event_type: &str) {
        self.business_events_total
            .with_label_values(&[domain, event_type])
            .inc();
    }

    fn render(&self) -> String {
        let encoder = TextEncoder::new();
        let metric_families = self.registry.gather();
        let mut buffer = Vec::new();
        encoder.encode(&metric_families, &mut buffer).unwrap();
        String::from_utf8(buffer).unwrap()
    }
}

// =============================================================================
// Structured Logging (generated by LDD-Kit adapt.py)
// =============================================================================
fn setup_logging() {
    let filter = EnvFilter::try_from_default_env()
        .unwrap_or_else(|_| EnvFilter::new("info"));

    if log_format() == "json" {
        tracing_subscriber::registry()
            .with(filter)
            .with(tracing_subscriber::fmt::layer().json())
            .init();
    } else {
        tracing_subscriber::registry()
            .with(filter)
            .with(tracing_subscriber::fmt::layer().pretty())
            .init();
    }
}

fn log_event(event_type: &str, fields: &serde_json::Map<String, serde_json::Value>) {
    let mut entry = json!({
        "event_type": event_type,
        "timestamp": Utc::now().to_rfc3339(),
        "service": SERVICE_NAME_VALUE,
    });

    if let Some(obj) = entry.as_object_mut() {
        for (k, v) in fields {
            obj.insert(k.clone(), v.clone());
        }
    }

    info!(target: "ldd_event", "{}", entry.to_string());
}

// =============================================================================
// OpenTelemetry Tracing Setup (generated by LDD-Kit adapt.py)
// =============================================================================
fn setup_tracing() -> trace::TracerProvider {
    global::set_text_map_propagator(TraceContextPropagator::new());

    let otlp_endpoint = env::var("OTEL_EXPORTER_OTLP_ENDPOINT").ok();

    let mut exporter_builder = opentelemetry_otlp::new_exporter().tonic();
    if let Some(endpoint) = otlp_endpoint {
        exporter_builder = exporter_builder.with_endpoint(endpoint);
    }

    let tracer_provider = opentelemetry_otlp::new_pipeline()
        .tracing()
        .with_exporter(exporter_builder)
        .with_trace_config(
            trace::Config::default()
                .with_sampler(Sampler::AlwaysOn)
                .with_id_generator(RandomIdGenerator::default())
                .with_resource(Resource::new(vec![KeyValue::new(
                    SERVICE_NAME,
                    otel_service_name(),
                )])),
        )
        .install_batch(Tokio)
        .expect("Failed to create tracer provider");

    global::set_tracer_provider(tracer_provider.clone());
    tracer_provider
}

// =============================================================================
// Telemetry Middleware (generated by LDD-Kit adapt.py)
// =============================================================================
async fn telemetry_middleware(
    State(metrics): State<Arc<Metrics>>,
    request: Request,
    next: Next,
) -> Response {
    let start = Instant::now();
    let method = request.method().to_string();
    let path = request.uri().path().to_string();

    let tracer = global::tracer(SERVICE_NAME_VALUE);
    let mut span = tracer.start(format!("{} {}", method, path));
    span.set_attribute(KeyValue::new("http.method", method.clone()));
    span.set_attribute(KeyValue::new("http.route", path.clone()));

    let cx = Context::current_with_span(span);

    let response = next.run(request).await;

    let duration = start.elapsed();
    let status_code = response.status().as_u16();

    span = cx.span().clone();
    span.set_attribute(KeyValue::new("http.status_code", status_code as i64));
    span.set_attribute(KeyValue::new("http.duration_ms", duration.as_millis() as f64));
    span.end();

    metrics.record_request(&method, &path, status_code, duration.as_secs_f64());

    info!(
        method = %method,
        path = %path,
        status_code = status_code,
        duration_ms = duration.as_millis() as f64,
        "http_request"
    );

    response
}

// =============================================================================
// Shared application state
// =============================================================================
#[derive(Clone)]
struct AppState {
    metrics: Arc<Metrics>,
}

// =============================================================================
// Handlers
// =============================================================================
async fn health() -> impl IntoResponse {
    Json(json!({"status": "ok", "service": SERVICE_NAME_VALUE}))
}

async fn ready() -> impl IntoResponse {
    Json(json!({"status": "ready"}))
}

async fn metrics_handler(State(state): State<AppState>) -> impl IntoResponse {
    (
        StatusCode::OK,
        [("content-type", "text/plain; version=0.0.4")],
        state.metrics.render(),
    )
}

#[derive(Deserialize)]
struct CreateUserRequest {
    user_id: String,
    email: String,
    #[serde(default = "default_tier")]
    tier: String,
}

fn default_tier() -> String {
    "free".to_string()
}

async fn create_user(
    State(state): State<AppState>,
    Json(body): Json<CreateUserRequest>,
) -> impl IntoResponse {
    let tracer = global::tracer(SERVICE_NAME_VALUE);
    let mut span = tracer.start("create_user");
    span.set_attribute(KeyValue::new("user.id", body.user_id.clone()));

    info!(
        event_type = "user.creating",
        user_id = %body.user_id,
        email = %body.email,
        tier = %body.tier,
        "user.creating"
    );

    // Simulate business logic
    tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;

    let event = UserCreated::new(body.user_id.clone(), body.email, body.tier);
    let event_json = serde_json::to_value(&event).unwrap();
    if let Some(obj) = event_json.as_object() {
        log_event("user.created", obj);
    }
    state.metrics.record_business_event("user", "user.created");

    span.end();

    (
        StatusCode::CREATED,
        Json(json!({
            "user_id": event.user_id,
            "status": "created",
            "tier": event.tier,
        })),
    )
}

async fn delete_user(
    State(state): State<AppState>,
    Path(user_id): Path<String>,
) -> impl IntoResponse {
    let event = UserDeleted::new(user_id.clone(), Some("user_request".to_string()));
    let event_json = serde_json::to_value(&event).unwrap();
    if let Some(obj) = event_json.as_object() {
        log_event("user.deleted", obj);
    }
    state.metrics.record_business_event("user", "user.deleted");

    Json(json!({"user_id": user_id, "status": "deleted"}))
}

#[derive(Deserialize)]
struct ProcessPaymentRequest {
    payment_id: String,
    amount_cents: i64,
    #[serde(default = "default_currency")]
    currency: String,
}

fn default_currency() -> String {
    "USD".to_string()
}

async fn process_payment(
    State(state): State<AppState>,
    Json(body): Json<ProcessPaymentRequest>,
) -> impl IntoResponse {
    let tracer = global::tracer(SERVICE_NAME_VALUE);
    let mut span = tracer.start("process_payment");
    span.set_attribute(KeyValue::new("payment.id", body.payment_id.clone()));
    span.set_attribute(KeyValue::new("payment.amount_cents", body.amount_cents));

    info!(
        event_type = "payment.processing",
        payment_id = %body.payment_id,
        amount_cents = body.amount_cents,
        "payment.processing"
    );

    // Simulate processing
    tokio::time::sleep(tokio::time::Duration::from_millis(20)).await;

    let event = PaymentProcessed::new(
        body.payment_id.clone(),
        body.amount_cents,
        body.currency,
        "success".to_string(),
    );
    let event_json = serde_json::to_value(&event).unwrap();
    if let Some(obj) = event_json.as_object() {
        log_event("payment.processed", obj);
    }
    state.metrics.record_business_event("payment", "payment.processed");

    span.end();

    Json(json!({
        "payment_id": event.payment_id,
        "status": "processed",
        "amount_cents": event.amount_cents,
    }))
}

async fn get_user(Path(user_id): Path<String>) -> impl IntoResponse {
    info!(
        event_type = "user.fetching",
        user_id = %user_id,
        "user.fetching"
    );

    Json(json!({
        "user_id": user_id,
        "email": format!("user_{}@example.com", user_id),
        "tier": "free",
    }))
}

// =============================================================================
// Main
// =============================================================================
#[tokio::main]
async fn main() {
    setup_logging();

    let _tracer_provider = if telemetry_enabled() {
        Some(setup_tracing())
    } else {
        None
    };

    let metrics = Arc::new(Metrics::new().expect("Failed to create metrics"));

    let state = AppState {
        metrics: metrics.clone(),
    };

    let app = Router::new()
        .route("/health", get(health))
        .route("/ready", get(ready))
        .route("/metrics", get(metrics_handler))
        .route("/api/v1/users", post(create_user))
        .route("/api/v1/users/:user_id", delete(delete_user).get(get_user))
        .route("/api/v1/payments", post(process_payment))
        .layer(middleware::from_fn_with_state(
            metrics.clone(),
            telemetry_middleware,
        ))
        .layer(TraceLayer::new_for_http())
        .with_state(state);

    let addr: SocketAddr = env::var("BIND_ADDR")
        .unwrap_or_else(|_| "0.0.0.0:8080".to_string())
        .parse()
        .expect("Invalid bind address");

    info!(
        event_type = "service_starting",
        service = SERVICE_NAME_VALUE,
        address = %addr,
        "service_starting"
    );

    let listener = tokio::net::TcpListener::bind(addr).await.unwrap();
    axum::serve(listener, app).await.unwrap();
}
```

## Running the Example

```bash
# Development mode (console logs, 100% trace sampling)
export MY_SERVICE_TELEMETRY_ENABLED=true
export MY_SERVICE_LOG_FORMAT=console
export OTEL_SERVICE_NAME=axum-example-local
export OTEL_TRACES_SAMPLER=always_on

cargo run

# In another terminal, test the endpoints:
curl http://localhost:8080/health
curl -X POST http://localhost:8080/api/v1/users \
  -H "Content-Type: application/json" \
  -d '{"user_id":"usr-001","email":"alice@example.com","tier":"pro"}'
curl http://localhost:8080/metrics
curl -X POST http://localhost:8080/api/v1/payments \
  -H "Content-Type: application/json" \

```

## What to Observe

1. **Structured logs** in the terminal with `event_type`, `trace_id`, and `duration_ms`
2. **Trace spans** for each HTTP request and business operation
3. **Prometheus metrics** at `/metrics` showing request counts and durations
4. **Event schemas** validated for every business event (`user.created`, `payment.processed`)
