# Example: LDD-Kit with Node.js / Express

> Complete minimal example of using LDD-Kit with a Node.js Express (TypeScript) service.

## Prerequisites

- Node.js 18+
- `npm` or `yarn`

## Project Setup

```bash
# 1. Create project directory
mkdir nodejs-express-example && cd nodejs-express-example

# 2. Initialize Node.js project
npm init -y

# 3. Install dependencies
npm install express @opentelemetry/api @opentelemetry/sdk-node \
  @opentelemetry/auto-instrumentations-node \
  @opentelemetry/exporter-trace-otlp-grpc \
  @opentelemetry/sdk-trace-base prom-client winston

# 4. Install dev dependencies
npm install -D typescript @types/express @types/node ts-node nodemon

# 5. Create tsconfig.json
cat > tsconfig.json << 'EOF'
{
  "compilerOptions": {
    "target": "ES2020",
    "module": "commonjs",
    "lib": ["ES2020"],
    "outDir": "./dist",
    "rootDir": "./src",
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    "resolveJsonModule": true,
    "declaration": true,
    "declarationMap": true,
    "sourceMap": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist"]
}
EOF
```

## Files

### `src/app.ts` — Complete Minimal Express App with Telemetry

```typescript
/**
 * Minimal Express (TypeScript) application with LDD-Kit telemetry.
 *
 * This demonstrates:
 *   - Telemetry setup and initialization
 *   - Telemetry middleware for automatic HTTP request tracing
 *   - Structured logging with event schemas
 *   - Prometheus metrics endpoint
 *   - Trace context propagation
 *
 * Environment variables:
 *     MY_SERVICE_TELEMETRY_ENABLED=true|false
 *     MY_SERVICE_LOG_FORMAT=json|console
 *     OTEL_SERVICE_NAME=my-service
 *     OTEL_TRACES_SAMPLER=always_on|parentbased_traceidratio
 */

import express, { Request, Response, NextFunction } from 'express';
import { NodeSDK } from '@opentelemetry/sdk-node';
import { OTLPTraceExporter } from '@opentelemetry/exporter-trace-otlp-grpc';
import { Resource } from '@opentelemetry/resources';
import { SemanticResourceAttributes } from '@opentelemetry/semantic-conventions';
import { getNodeAutoInstrumentations } from '@opentelemetry/auto-instrumentations-node';
import { trace, context, SpanStatusCode } from '@opentelemetry/api';
import * as promClient from 'prom-client';
import winston from 'winston';

// =============================================================================
// Configuration
// =============================================================================
const SERVICE_NAME = 'nodejs-express-example';
const TELEMETRY_ENABLED = (process.env.MY_SERVICE_TELEMETRY_ENABLED || 'false').toLowerCase() === 'true';
const LOG_FORMAT = process.env.MY_SERVICE_LOG_FORMAT || 'console';
const OTEL_SERVICE_NAME = process.env.OTEL_SERVICE_NAME || SERVICE_NAME;
const PORT = parseInt(process.env.PORT || '3000', 10);

// =============================================================================
// Structured Logging Setup (generated by LDD-Kit adapt.py)
// =============================================================================
const winstonFormat = LOG_FORMAT === 'json'
  ? winston.format.combine(
      winston.format.timestamp(),
      winston.format.errors({ stack: true }),
      winston.format.json()
    )
  : winston.format.combine(
      winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss' }),
      winston.format.colorize(),
      winston.format.printf(({ level, message, timestamp, ...rest }) => {
        const extras = Object.keys(rest).length > 0 ? JSON.stringify(rest) : '';
        return `[${timestamp}] ${level}: ${message} ${extras}`;
      })
    );

const logger = winston.createLogger({
  level: 'info',
  defaultMeta: { service: SERVICE_NAME },
  format: winstonFormat,
  transports: [new winston.transports.Console()],
});

// Helper to inject trace context into log entries
function createLogMeta(req?: Request): Record<string, unknown> {
  const span = trace.getActiveSpan();
  const meta: Record<string, unknown> = {};
  if (span) {
    const ctx = span.spanContext();
    meta.trace_id = ctx.traceId;
    meta.span_id = ctx.spanId;
  }
  if (req) {
    meta.request_id = (req as any).requestId;
  }
  return meta;
}

// =============================================================================
// OpenTelemetry Tracing Setup (generated by LDD-Kit adapt.py)
// =============================================================================
let otelSDK: NodeSDK | null = null;

function setupTracing(): NodeSDK | null {
  if (!TELEMETRY_ENABLED) return null;

  const traceExporter = new OTLPTraceExporter({
    url: process.env.OTEL_EXPORTER_OTLP_ENDPOINT,
  });

  const sdk = new NodeSDK({
    traceExporter,
    instrumentations: [getNodeAutoInstrumentations()],
    resource: new Resource({
      [SemanticResourceAttributes.SERVICE_NAME]: OTEL_SERVICE_NAME,
      [SemanticResourceAttributes.SERVICE_VERSION]: '1.0.0',
    }),
  });

  sdk.start();
  return sdk;
}

function shutdownTracing(): Promise<void> {
  if (otelSDK) {
    return otelSDK.shutdown().catch((err) => {
      logger.error('Error shutting down OpenTelemetry', { error: err.message });
    });
  }
  return Promise.resolve();
}

otelSDK = setupTracing();
const tracer = trace.getTracer(SERVICE_NAME);

// =============================================================================
// Prometheus Metrics (generated by LDD-Kit adapt.py)
// =============================================================================
const httpRequestsTotal = new promClient.Counter({
  name: 'express_example_http_requests_total',
  help: 'Total HTTP requests',
  labelNames: ['method', 'path', 'status_code'],
});

const httpRequestDuration = new promClient.Histogram({
  name: 'express_example_http_request_duration_seconds',
  help: 'HTTP request duration in seconds',
  labelNames: ['method', 'path'],
  buckets: [0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0],
});

const businessEventsTotal = new promClient.Counter({
  name: 'express_example_business_events_total',
  help: 'Total business events',
  labelNames: ['domain', 'event_type'],
});

// =============================================================================
// Event Schemas (generated by LDD-Kit adapt.py)
// =============================================================================
interface BaseEvent {
  event_type: string;
  event_domain: string;
  timestamp: string;
}

class UserCreated implements BaseEvent {
  event_type = 'user.created' as const;
  event_domain = 'user' as const;
  timestamp = new Date().toISOString();

  constructor(
    public user_id: string,
    public email: string,
    public tier: string = 'free'
  ) {}

  toLogFields(): Record<string, unknown> {
    return {
      event_type: this.event_type,
      event_domain: this.event_domain,
      timestamp: this.timestamp,
      user_id: this.user_id,
      email: this.email,
      tier: this.tier,
    };
  }
}

class UserDeleted implements BaseEvent {
  event_type = 'user.deleted' as const;
  event_domain = 'user' as const;
  timestamp = new Date().toISOString();

  constructor(
    public user_id: string,
    public reason?: string
  ) {}

  toLogFields(): Record<string, unknown> {
    return {
      event_type: this.event_type,
      event_domain: this.event_domain,
      timestamp: this.timestamp,
      user_id: this.user_id,
      reason: this.reason,
    };
  }
}

class PaymentProcessed implements BaseEvent {
  event_type = 'payment.processed' as const;
  event_domain = 'payment' as const;
  timestamp = new Date().toISOString();

  constructor(
    public payment_id: string,
    public amount_cents: number,
    public currency: string = 'USD',
    public status: string = 'success'
  ) {}

  toLogFields(): Record<string, unknown> {
    return {
      event_type: this.event_type,
      event_domain: this.event_domain,
      timestamp: this.timestamp,
      payment_id: this.payment_id,
      amount_cents: this.amount_cents,
      currency: this.currency,
      status: this.status,
    };
  }
}

// =============================================================================
// Telemetry Middleware (generated by LDD-Kit adapt.py)
// =============================================================================
function telemetryMiddleware(req: Request, res: Response, next: NextFunction): void {
  const start = Date.now();
  const path = req.path;
  const method = req.method;
  const requestId = `req-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
  (req as any).requestId = requestId;

  // Start a trace span for the request
  const span = tracer.startSpan('http_request', {
    attributes: {
      'http.method': method,
      'http.route': path,
      'http.url': req.originalUrl,
    },
  });

  // Store the active context
  const ctx = trace.setSpan(context.active(), span);
  context.with(ctx, () => {
    res.on('finish', () => {
      const duration = Date.now() - start;
      const statusCode = res.statusCode.toString();

      httpRequestsTotal.inc({ method, path, status_code: statusCode });
      httpRequestDuration.observe({ method, path }, duration / 1000);

      span.setAttribute('http.status_code', res.statusCode);
      span.setAttribute('http.duration_ms', duration);

      if (res.statusCode >= 500) {
        span.setStatus({ code: SpanStatusCode.ERROR });
      }
      span.end();

      logger.info('http_request', {
        ...createLogMeta(req),
        method,
        path,
        status_code: res.statusCode,
        duration_ms: duration,
      });
    });

    next();
  });
}

// =============================================================================
// Express Application
// =============================================================================
const app = express();
app.use(express.json());
app.use(telemetryMiddleware);

// Health check
app.get('/health', (req: Request, res: Response) => {
  res.json({ status: 'ok', service: SERVICE_NAME });
});

// Readiness probe
app.get('/ready', (req: Request, res: Response) => {
  res.json({ status: 'ready' });
});

// Prometheus metrics
app.get('/metrics', async (req: Request, res: Response) => {
  res.set('Content-Type', promClient.register.contentType);
  res.end(await promClient.register.metrics());
});

// Business routes
app.post('/api/v1/users', async (req: Request, res: Response) => {
  const { user_id, email, tier = 'free' } = req.body;

  await tracer.startActiveSpan('create_user', async (span) => {
    span.setAttribute('user.id', user_id);

    logger.info('user.creating', { ...createLogMeta(req), user_id, email, tier });

    // Simulate business logic
    await new Promise(resolve => setTimeout(resolve, 10));

    const event = new UserCreated(user_id, email, tier);
    logger.info('user.created', { ...createLogMeta(req), ...event.toLogFields() });
    businessEventsTotal.inc({ domain: 'user', event_type: 'user.created' });

    span.end();
  });

  res.status(201).json({ user_id, status: 'created', tier });
});

app.delete('/api/v1/users/:user_id', async (req: Request, res: Response) => {
  const { user_id } = req.params;

  const event = new UserDeleted(user_id, 'user_request');
  logger.info('user.deleted', { ...createLogMeta(req), ...event.toLogFields() });
  businessEventsTotal.inc({ domain: 'user', event_type: 'user.deleted' });

  res.json({ user_id, status: 'deleted' });
});

app.post('/api/v1/payments', async (req: Request, res: Response) => {
  const { payment_id, amount_cents, currency = 'USD' } = req.body;

  await tracer.startActiveSpan('process_payment', async (span) => {
    span.setAttribute('payment.id', payment_id);
    span.setAttribute('payment.amount_cents', amount_cents);

    logger.info('payment.processing', { ...createLogMeta(req), payment_id, amount_cents });

    // Simulate processing
    await new Promise(resolve => setTimeout(resolve, 20));

    const event = new PaymentProcessed(payment_id, amount_cents, currency, 'success');
    logger.info('payment.processed', { ...createLogMeta(req), ...event.toLogFields() });
    businessEventsTotal.inc({ domain: 'payment', event_type: 'payment.processed' });

    span.end();
  });

  res.json({ payment_id, status: 'processed', amount_cents });
});

app.get('/api/v1/users/:user_id', (req: Request, res: Response) => {
  const { user_id } = req.params;

  logger.info('user.fetching', { ...createLogMeta(req), user_id });

  res.json({
    user_id,
    email: `user_${user_id}@example.com`,
    tier: 'free',
  });
});

// Graceful shutdown
process.on('SIGTERM', async () => {
  logger.info('service_stopping', { service: SERVICE_NAME });
  await shutdownTracing();
  process.exit(0);
});

// Start server
app.listen(PORT, () => {
  logger.info('service_starting', { service: SERVICE_NAME, port: PORT });
});

export { app };
```

## Running the Example

```bash
# Development mode (console logs, 100% trace sampling)
export MY_SERVICE_TELEMETRY_ENABLED=true
export MY_SERVICE_LOG_FORMAT=console
export OTEL_SERVICE_NAME=express-example-local
export OTEL_TRACES_SAMPLER=always_on

npm run dev

# In another terminal, test the endpoints:
curl http://localhost:3000/health
curl -X POST http://localhost:3000/api/v1/users \
  -H "Content-Type: application/json" \
  -d '{"user_id":"usr-001","email":"alice@example.com","tier":"pro"}'
curl http://localhost:3000/metrics
curl -X POST http://localhost:3000/api/v1/payments \
  -H "Content-Type: application/json" \
  -d '{"payment_id":"pay-001","amount_cents":999,"currency":"USD"}'
```

## What to Observe

1. **Structured logs** in the terminal with `event_type`, `trace_id`, and `duration_ms`
2. **Trace spans** for each HTTP request and business operation
3. **Prometheus metrics** at `/metrics` showing request counts and durations
4. **Event schemas** validated for every business event (`user.created`, `payment.processed`)
