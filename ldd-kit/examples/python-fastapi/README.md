# Example: LDD-Kit with Python / FastAPI

> Complete minimal example of using LDD-Kit with a Python FastAPI service.

## Prerequisites

- Python 3.11+
- `pip` or `uv`

## Project Setup

```bash
# 1. Create project directory
mkdir python-fastapi-example && cd python-fastapi-example

# 2. Create virtual environment
python -m venv .venv
source .venv/bin/activate

# 3. Install dependencies
pip install fastapi uvicorn structlog opentelemetry-api opentelemetry-sdk \
  opentelemetry-instrumentation-fastapi opentelemetry-exporter-otlp \
  prometheus-client pydantic
```

## Files

### `main.py` — Complete Minimal FastAPI App with Telemetry

```python
"""
Minimal FastAPI application with LDD-Kit telemetry.

This demonstrates:
  - Telemetry setup and initialization
  - Telemetry middleware for automatic HTTP request tracing
  - Structured logging with event schemas
  - Prometheus metrics endpoint
  - Trace context propagation

Environment variables:
    MY_SERVICE_TELEMETRY_ENABLED=true|false
    MY_SERVICE_LOG_FORMAT=json|console
    OTEL_SERVICE_NAME=my-service
    OTEL_TRACES_SAMPLER=always_on|parentbased_traceidratio
"""

import os
import time
from contextlib import asynccontextmanager
from datetime import datetime
from typing import Optional

from fastapi import FastAPI, Request, Response
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import PlainTextResponse, JSONResponse
from pydantic import BaseModel, Field

# =============================================================================
# Telemetry imports (generated by LDD-Kit adapt.py)
# =============================================================================
# In a real project, these come from the generated telemetry package:
#   from my_service.telemetry import setup_telemetry, get_logger, TelemetryMiddleware
# For this minimal example, we inline the telemetry setup.
# =============================================================================

import structlog
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import (
    BatchSpanProcessor,
    ConsoleSpanExporter,
)
from opentelemetry.sdk.resources import Resource, SERVICE_NAME, SERVICE_VERSION
from opentelemetry.instrumentation.fastapi import FastAPIInstrumentor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from prometheus_client import Counter, Histogram, generate_latest, CONTENT_TYPE_LATEST


# =============================================================================
# Configuration
# =============================================================================
SERVICE_NAME = "python-fastapi-example"
TELEMETRY_ENABLED = os.environ.get("MY_SERVICE_TELEMETRY_ENABLED", "false").lower() == "true"
LOG_FORMAT = os.environ.get("MY_SERVICE_LOG_FORMAT", "console")
OTEL_SERVICE_NAME = os.environ.get("OTEL_SERVICE_NAME", SERVICE_NAME)


# =============================================================================
# Structured Logging Setup
# =============================================================================
def setup_logging():
    """Configure structlog for structured JSON or console output."""
    processors = [
        structlog.contextvars.merge_contextvars,
        structlog.processors.add_log_level,
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.stdlib.ExtraAdder(),
    ]
    if LOG_FORMAT == "json":
        processors.append(structlog.processors.JSONRenderer())
    else:
        processors.append(structlog.dev.ConsoleRenderer(colors=True))

    structlog.configure(
        processors=processors,
        wrapper_class=structlog.make_filtering_bound_logger(20),  # INFO
        context_class=dict,
        logger_factory=structlog.PrintLoggerFactory(),
    )


setup_logging()
logger = structlog.get_logger(SERVICE_NAME)


# =============================================================================
# OpenTelemetry Tracing Setup
# =============================================================================
def setup_tracing():
    """Configure OpenTelemetry tracing with OTLP or console fallback."""
    resource = Resource.create({
        SERVICE_NAME: OTEL_SERVICE_NAME,
        SERVICE_VERSION: "1.0.0",
    })
    provider = TracerProvider(resource=resource)

    otlp_endpoint = os.environ.get("OTEL_EXPORTER_OTLP_ENDPOINT")
    if otlp_endpoint:
        exporter = OTLPSpanExporter(endpoint=otlp_endpoint)
    else:
        exporter = ConsoleSpanExporter()

    processor = BatchSpanProcessor(exporter)
    provider.add_span_processor(processor)
    trace.set_tracer_provider(provider)
    return provider


tracer_provider: Optional[TracerProvider] = None
if TELEMETRY_ENABLED:
    tracer_provider = setup_tracing()

tracer = trace.get_tracer(SERVICE_NAME)


# =============================================================================
# Prometheus Metrics
# =============================================================================
http_requests_total = Counter(
    "fastapi_example_http_requests_total",
    "Total HTTP requests",
    ["method", "path", "status_code"],
)

http_request_duration = Histogram(
    "fastapi_example_http_request_duration_seconds",
    "HTTP request duration in seconds",
    ["method", "path"],
    buckets=[0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0],
)

business_events_total = Counter(
    "fastapi_example_business_events_total",
    "Total business events",
    ["domain", "event_type"],
)


# =============================================================================
# Event Schemas (generated by LDD-Kit adapt.py)
# =============================================================================
class UserCreated(BaseModel):
    """Event emitted when a new user is created."""
    event_type: str = Field(default="user.created", const=True)
    event_domain: str = Field(default="user", const=True)
    timestamp: str = Field(default_factory=lambda: datetime.utcnow().isoformat() + "Z")
    user_id: str
    email: str
    tier: str = "free"


class UserDeleted(BaseModel):
    """Event emitted when a user is deleted."""
    event_type: str = Field(default="user.deleted", const=True)
    event_domain: str = Field(default="user", const=True)
    timestamp: str = Field(default_factory=lambda: datetime.utcnow().isoformat() + "Z")
    user_id: str
    reason: Optional[str] = None


class PaymentProcessed(BaseModel):
    """Event emitted when a payment is processed."""
    event_type: str = Field(default="payment.processed", const=True)
    event_domain: str = Field(default="payment", const=True)
    timestamp: str = Field(default_factory=lambda: datetime.utcnow().isoformat() + "Z")
    payment_id: str
    amount_cents: int
    currency: str = "USD"
    status: str = "success"


# =============================================================================
# Telemetry Middleware (generated by LDD-Kit adapt.py)
# =============================================================================
class TelemetryMiddleware:
    """ASGI middleware for automatic HTTP request telemetry."""

    def __init__(self, app):
        self.app = app

    async def __call__(self, scope, receive, send):
        if scope["type"] != "http":
            await self.app(scope, receive, send)
            return

        method = scope["method"]
        path = scope.get("path", "/")
        start_time = time.perf_counter()

        # Capture response status
        status_code = 200
        async def wrapped_send(message):
            nonlocal status_code
            if message["type"] == "http.response.start":
                status_code = message.get("status", 200)
            await send(message)

        with tracer.start_as_current_span("http_request") as span:
            span.set_attribute("http.method", method)
            span.set_attribute("http.route", path)

            await self.app(scope, receive, wrapped_send)

            duration = time.perf_counter() - start_time
            http_requests_total.labels(
                method=method, path=path, status_code=str(status_code)
            ).inc()
            http_request_duration.labels(
                method=method, path=path
            ).observe(duration)
            span.set_attribute("http.status_code", status_code)
            span.set_attribute("http.duration_ms", duration * 1000)

            logger.info(
                "http_request",
                method=method,
                path=path,
                status_code=status_code,
                duration_ms=round(duration * 1000, 2),
                trace_id=format(span.get_span_context().trace_id, "032x"),
            )


# =============================================================================
# FastAPI Application
# =============================================================================
@asynccontextmanager
async def lifespan(app: FastAPI):
    """Application lifespan handler."""
    logger.info("service_starting", service=SERVICE_NAME, port=8000)
    yield
    logger.info("service_stopping", service=SERVICE_NAME)


app = FastAPI(
    title="LDD-Kit Python/FastAPI Example",
    version="1.0.0",
    lifespan=lifespan,
)

# Add middlewares
app.add_middleware(TelemetryMiddleware)

# Instrument with OpenTelemetry
if TELEMETRY_ENABLED:
    FastAPIInstrumentor.instrument_app(app)


# =============================================================================
# Routes
# =============================================================================

@app.get("/health")
async def health():
    """Health check endpoint."""
    return {"status": "ok", "service": SERVICE_NAME}


@app.get("/ready")
async def ready():
    """Readiness probe."""
    return {"status": "ready"}


@app.get("/metrics")
async def metrics():
    """Prometheus metrics endpoint."""
    return PlainTextResponse(
        content=generate_latest().decode("utf-8"),
        media_type=CONTENT_TYPE_LATEST,
    )


@app.post("/api/v1/users")
async def create_user(request: Request):
    """Create a new user — demonstrates structured event logging."""
    body = await request.json()
    user_id = body.get("user_id", "unknown")
    email = body.get("email", "unknown")
    tier = body.get("tier", "free")

    with tracer.start_as_current_span("create_user") as span:
        span.set_attribute("user.id", user_id)
        span.set_attribute("user.email", email)

        logger.info("user.creating", user_id=user_id, email=email, tier=tier)

        # Simulate business logic
        time.sleep(0.01)

        event = UserCreated(user_id=user_id, email=email, tier=tier)
        logger.info("user.created", **event.model_dump())
        business_events_total.labels(domain="user", event_type="user.created").inc()

    return {"user_id": user_id, "status": "created", "tier": tier}


@app.delete("/api/v1/users/{user_id}")
async def delete_user(user_id: str):
    """Delete a user — demonstrates event logging with optional fields."""
    with tracer.start_as_current_span("delete_user"):
        event = UserDeleted(user_id=user_id, reason="user_request")
        logger.info("user.deleted", **event.model_dump())
        business_events_total.labels(domain="user", event_type="user.deleted").inc()

    return {"user_id": user_id, "status": "deleted"}


@app.post("/api/v1/payments")
async def process_payment(request: Request):
    """Process a payment — demonstrates payment domain events."""
    body = await request.json()
    payment_id = body.get("payment_id", "unknown")
    amount_cents = body.get("amount_cents", 0)
    currency = body.get("currency", "USD")

    with tracer.start_as_current_span("process_payment") as span:
        span.set_attribute("payment.id", payment_id)
        span.set_attribute("payment.amount_cents", amount_cents)

        logger.info("payment.processing", payment_id=payment_id, amount_cents=amount_cents)

        # Simulate processing
        time.sleep(0.02)

        event = PaymentProcessed(
            payment_id=payment_id,
            amount_cents=amount_cents,
            currency=currency,
            status="success",
        )
        logger.info("payment.processed", **event.model_dump())
        business_events_total.labels(domain="payment", event_type="payment.processed").inc()

    return {"payment_id": payment_id, "status": "processed", "amount_cents": amount_cents}


@app.get("/api/v1/users/{user_id}")
async def get_user(user_id: str):
    """Get user details — demonstrates simple trace span."""
    with tracer.start_as_current_span("get_user"):
        logger.info("user.fetching", user_id=user_id)
        return {
            "user_id": user_id,
            "email": f"user_{user_id}@example.com",
            "tier": "free",
        }


# =============================================================================
# Main entry point
# =============================================================================
if __name__ == "__main__":
    import uvicorn
    uvicorn.run(
        "main:app",
        host="0.0.0.0",
        port=int(os.environ.get("PORT", 8000)),
        reload=os.environ.get("ENV", "production") == "development",
    )
```

## Running the Example

```bash
# Development mode (console logs, 100% trace sampling)
export MY_SERVICE_TELEMETRY_ENABLED=true
export MY_SERVICE_LOG_FORMAT=console
export OTEL_SERVICE_NAME=fastapi-example-local
export OTEL_TRACES_SAMPLER=always_on

uvicorn main:app --reload --port 8000

# In another terminal, test the endpoints:
curl http://localhost:8000/health
curl -X POST http://localhost:8000/api/v1/users \
  -H "Content-Type: application/json" \
  -d '{"user_id":"usr-001","email":"alice@example.com","tier":"pro"}'
curl http://localhost:8000/metrics
curl -X POST http://localhost:8000/api/v1/payments \
  -H "Content-Type: application/json" \
  -d '{"payment_id":"pay-001","amount_cents":999,"currency":"USD"}'
```

## What to Observe

1. **Structured logs** in the terminal with `event_type`, `trace_id`, and `duration_ms`
2. **Trace spans** for each HTTP request and business operation
3. **Prometheus metrics** at `/metrics` showing request counts and durations
4. **Event schemas** validated for every business event (`user.created`, `payment.processed`)
